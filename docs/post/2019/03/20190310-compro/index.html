類題は https://atcoder.jp/contests/abc120/tasks/abc120_c とか https://atcoder.jp/contests/agc005/tasks/agc005_a 。

クソコード書いて提出。もっといいやり方ある気がしてならない。

signed main() {
    stack<char> st;
    string s;
    cin >> s;
    for (auto c : s) {
        if (st.empty()) {
            st.push(c);
        } else {
            if (st.top() == 'W' && c == 'A') {
                int cnt = 0;
                while (!st.empty() && st.top() == 'W') {
                    st.pop();
                    cnt++;
                }
                st.push('A');
                for (int i = 0; i < cnt; i++) {
                    st.push('C');
                }
            } else {
                st.push(c);
            }
        }
    }
 
    string ans;
    while (!st.empty()) {
        char c = st.top();
        st.pop();
        ans += c;
    }
    reverse(ans.begin(), ans.end());
    cout << ans << endl;
}
